Qus:Write to print factorial of given number?
Ans:
<script>
function factorial(n) {
let answer = 1;
if (n == 0 || n == 1) {
return answer;
} else if (n > 1) {
for (var i = n; i >= 1; i--) {
answer = answer * i;
}
return answer;
} else {
return "number has to be positive.";
}
}
let n = parseInt(prompt('input digit for factorial'));
answer = factorial(n);
console.log("Factorial of " + n + " : " + answer);
</script>
