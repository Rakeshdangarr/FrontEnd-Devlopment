Qus:Write to check whether a number is negative, positive or zero?
Ans:
<script>
let checkNumber = parseInt(prompt("input Whether a number: "));

if (checkNumber > 0) {
console.log("number is positive");
}
else if (checkNumber == 0) {
console.log("number is Zero");
} else {
console.log("number is Negative");
}
</script>
